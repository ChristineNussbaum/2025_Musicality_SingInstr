#this is the stimuluslist. We load it via textfile

#(1)Filename(2)Duration (3)ID (4) Emo (5) Utterance (6) MType (7) Sex (8) FilenameSave (9)CorrCB1 (10)CorrCB2 (11)CorrCB3 (10)CorrCB4

#############################################################################
task setCB
  set &x random 1 4
  if &x == 1
    set &Task   slideTaskCB1
    set &Prac   slidePracCB1
    set %CB "CB1"
  fi
  if &x == 2
    set &Task   slideTaskCB2
    set &Prac   slidePracCB2
    set %CB "CB2"
  fi
  if &x == 3
    set &Task   slideTaskCB3
    set &Prac   slidePracCB3
    set %CB "CB3"
  fi
  if &x == 4
    set &Task   slideTaskCB4
    set &Prac   slidePracCB4
    set %CB "CB4"
  fi

#############################################################################

part EmoTask
  show bitmap slideFix
  delay 500
  keys d f j k                  # which keys are allowed
  sound @1
  delay @2
  if &x == 1                # CB1
  show bitmap LabelCB1 0 200
  readkey @9 5000
  fi 
  if &x == 2                # CB2
  show bitmap LabelCB2 0 200
  readkey @10 5000
  fi 
  if &x == 3                # CB3
  show bitmap LabelCB3 0 200
  readkey @11 5000
  fi 
  if &x == 4                # CB4
  show bitmap LabelCB4 0 200
  readkey @12 5000
  fi
  if STATUS == 3 #timeout
    show bitmap slideTooSlow
  fi
  if STATUS != 3 # alles andere
    show bitmap slideBlank
  fi
  delay 500
  save @3 @4 @5 @6 @7 @8 %CB STATUS KEY RT BLOCKNAME TABLEROW


task EmotionClassification
  table Stimuli
  part EmoTask


task PractiseClassification
  table Practise
  part EmoTask

task explanation1
  show bitmap &Task
  keys space
  readkey 1 100000

task explanation2
  show bitmap &Prac
  keys space
  readkey 1 100000
